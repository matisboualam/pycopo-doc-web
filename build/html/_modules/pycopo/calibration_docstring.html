    """Customly set markers size"""
    if np.isscalar(marker_coeff):
        marker_coeff = np.ones(compo.ids.shape[0]) * marker_coeff

    if inplace:
        compo = compo_metric
    else:
        return compo_metric




[docs]def graph_calibration(data, reference=None, weight="dist"):
    """Guesses the marker and camera pose using graph theory.
    :param data: The dataframe containing the marker data.
    :type data: pandas.DataFrame
    :param reference: the reference marker id. If None, the central marker of the
                 graph will be used. If not None, has to be the id of an
                 existing marker (experimental).
    :return: The camera pose and marker poses for each image label, as well as the graph and multigraph objects.
    :rtype: (pandas.DataFrame, pandas.DataFrame, networkx.Graph, networkx.MultiGraph)"""
    (marker_dicin, marker_dicout) = pcp.utils.string_map(
        data.markers.label.unique(), prefix="m"
    )
    (im_dicin, im_dicout) = pcp.utils.string_map(data.images.unique(), prefix="i")
    MG = pcp.utils.get_multigraph(data=data)
    G = pcp.utils.get_graph(data=data)
    if reference is None:
        centralMarker = pcp.utils.get_central_marker(data)
    else:
        centralMarker = reference
    markers = np.array(MG.nodes)
    nonCentralMarkers = set(markers) - set([centralMarker])
    cols = [
        ("rvec", "x"),
        ("rvec", "y"),
        ("rvec", "z"),
        ("tvec", "x"),
        ("tvec", "y"),
        ("tvec", "z"),
    ]
    cols = pd.MultiIndex.from_tuples(cols)
    mpose = pd.DataFrame(
        columns=cols, index=markers, data=np.zeros((len(markers), len(cols)))
    )
    mpose.index.name = "marker"
    for marker in nonCentralMarkers:
        (R, T) = pcp.utils.graph_path_RT(
            graph=G,
            data=data,
            nodeIn=marker_dicin[marker],
            nodeOut=marker_dicin[centralMarker],
        )
        mpose.loc[marker] = np.concatenate([R, T])
    mpose["central"] = False
    mpose.loc[centralMarker, "central"] = True
    imageLabels = data.images.unique()
    cpose = pd.DataFrame(
        columns=cols, index=imageLabels, data=np.zeros((len(imageLabels), len(cols)))
    )
    cpose.index.name = "imageLabel"
    for imageLabel in imageLabels:
        (R, T) = pcp.utils.graph_path_RT(
            graph=G,
            data=data,
            nodeIn=marker_dicin[centralMarker],
            nodeOut=im_dicin[imageLabel],
        )
        cpose.loc[imageLabel] = np.concatenate([R, T])
    return (cpose, mpose, G, MG)

    compo_target,
    targetPoseBatch,
    refPoseBatch=None,
    kind="point",
    fixed="z",
    more_features=False,
):
    """This function finds the geometrical feature of the composite based on target pose batch and the reference pose batch.

    :param targetPoseBatch: The target pose batch.
    :type targetPoseBatch: PoseBatch
    :param refPoseBatch: The reference pose batch.
    :type refPoseBatch: PoseBatch
    :param kind: The kind of geometrical feature.
    :type kind: str
    :param fixed: If the geometrical feature is fixed.
    :type fixed: bool
    :return: The geometrical feature of the target pose batch and the reference pose batch.
    :rtype: (rvec, tvec, residuals)"""
    (rvec, tvec, residuals) = pcp.calibration.find_geometrical_feature(
        targetPoseBatch=targetPoseBatch,
        refPoseBatch=refPoseBatch,
        kind=kind,
        fixed=fixed,
    )
    if kind == "point":
        (rvec_compo2newcompo, tvec_compo2newcompo) = pcp.utils.invert_RT(R=rvec, T=tvec)
    if kind == "axis":
        rvec_compo2newcompo = rvec
        tvec_compo2newcompo = tvec
    if kind == "bit":
        rvec_compo2newcompo = rvec
        tvec_compo2newcompo = tvec
    new_compo = compo_target.apply_RT(rvec_compo2newcompo, tvec_compo2newcompo)
    if not more_features:
        return new_compo
    else:
        return (new_compo, rvec_compo2newcompo, tvec_compo2newcompo)

    axis_dict = {"x": 0, "y": 1, "z": 2}
    if refPoseBatch is None:

        def unpack(X):
            """This function takes in an array of parameters X and returns the corresponding values for the rotation matrix R_target2a, the translation vector T_target2a, the rotation matrix R_b2cam, the translation vector T_b2cam, and the theta angle.

            :param X: The array of parameters.
            :type X: numpy.ndarray
            :return: The rotation matrix R_target2a, the translation vector T_target2a, the rotation matrix R_b2cam, the translation vector T_b2cam, and the theta angle.
            :rtype: tuple"""
            R_target2a = X[:3]
            T_target2a = np.zeros(3)
            T_target2a[axis_dict[fixed] - 1] = X[3]
            R_b2cam = X[4:7]
            T_b2cam = X[7:10]
            theta = X[10:]
            return (R_target2a, T_target2a, R_b2cam, T_b2cam, theta)

        def cost(X, R_cam2target, T_cam2target, axis):
            """This function unpacks the parameters of the camera-target-object system.

            :param X: The parameters of the camera-target-object system.
            :type X: np.array
            :return: The unpacked parameters of the camera-target-object system.
            :rtype: tuple"""
            (R_target2a, T_target2a, R_b2cam, T_b2cam, theta) = unpack(X)
            out = np.zeros((Nim, 6))
            for i in range(Nim):
                R_a2b = np.zeros(3)
                R_a2b[axis] = theta[i]
                T_a2b = np.zeros(3)
                (R_target2b, T_target2b) = pcp.utils.compose_RT(
                    R_target2a, T_target2a, R_a2b, T_a2b
                )
                (R_target2cam, T_target2cam) = pcp.utils.compose_RT(
                    R_target2b, T_target2b, R_b2cam, T_b2cam
                )
                (R_target2target, T_target2target) = pcp.utils.compose_RT(
                    R_target2cam, T_target2cam, R_cam2target[i], T_cam2target[i]
                )
                out[i, :3] = R_target2target
                out[i, 3:] = T_target2target
                print(
                    "optimizing\t",
                    datetime.now().strftime("%H:%M:%S"),
                    f"\tmean resisdu = {out.mean()}",
                    end="\r",
                )
            return out.flatten()

        data = targetPoseBatch
        R_cam2target = np.zeros((Nim, 3))
        T_cam2target = np.zeros((Nim, 3))
        for i in range(Nim):
            (R_cam2target[i], T_cam2target[i]) = pcp.utils.invert_RT(

[docs]def distance_batches(
    composite,
    path_0,
    path_1,
    parameters,
    camera_matrix,
    distorsion_coefficients,
    calib_type="cdist",
):
    """This function takes in the path to two images, a calibration type, and returns the error between the two images.

    :param path_0: The path to the first image.
    :type path_0: str
    :param path_1: The path to the second image.
    :type path_1: str
    :param calib_type: The type of calibration to use.
    :type calib_type: str
    :return: The error between the two images.
    :rtype: float"""
    batch_0 = pcp.calibration.ImageBatchCompositePose(
        composites={"target": composite},
        directory=path_0,
        parameters=parameters,
        camera_matrix=camera_matrix,
        distortion_coefficients=distorsion_coefficients,
    )
    batch_1 = pcp.calibration.ImageBatchCompositePose(
        composites={"target": composite},
        directory=path_1,
        parameters=parameters,
        camera_matrix=camera_matrix,
        distortion_coefficients=distorsion_coefficients,
    )
    batch_0.run(verbose=False)
    batch_1.run(verbose=False)
    pos_0 = batch_0.poseBatches["target"].tvecs
    pos_1 = batch_1.poseBatches["target"].tvecs
    if len(pos_0) > len(pos_1):
        pos_0 = pos_0[: len(pos_1)]
    else:
        pos_1 = pos_1[: len(pos_0)]
    if calib_type == "dist":
        return np.linalg.norm(pos_0 - pos_1, axis=1).flatten()
    if calib_type == "cdist":
        return cdist(pos_0, pos_1).flatten()

[docs]    def optimize_calibration(self, from_index=-1):
        """Optimizes the calibation using LM on reprojection error."""

        def expand_inputs(X, nonRefMarkers, compo, poseBatch):
            """
            Inverse of *flatten_inputs*
            """
            crvecs = poseBatch.rvecs
            ctvecs = poseBatch.tvecs
            loc = 0
            Nm = len(nonRefMarkers) - 1
            Ni = len(crvecs)
            compo.rvecs[nonRefMarkers] = X[loc : loc + Nm * 3].reshape(Nm, 3)
            loc += Nm * 3
            compo.tvecs[nonRefMarkers] = X[loc : loc + Nm * 3].reshape(Nm, 3)
            loc += Nm * 3
            crvecs[:] = X[loc : loc + Ni * 3].reshape(Ni, 3)
            loc += Ni * 3
            ctvecs[:] = X[loc : loc + Ni * 3].reshape(Ni, 3)

        def flatten_inputs(nonRefMarkers, compo, poseBatch):
            """This function concatenates the rvecs and tvecs of a poseBatch and returns the result.

            :param poseBatch: The poseBatch to concatenate.
            :type poseBatch: PoseBatch
            :param nonRefMarkers: The non-reference markers to use.
            :type nonRefMarkers: list of int
            :return: The concatenated rvecs and tvecs of the poseBatch.
            :rtype: np.array"""
            crvecs = poseBatch.rvecs
            ctvecs = poseBatch.tvecs
            return np.concatenate(
                [
                    compo.rvecs[nonRefMarkers].flatten(),
                    compo.tvecs[nonRefMarkers].flatten(),
                    crvecs.flatten(),
                    ctvecs.flatten(),
                ]
            )

        def cost_function(X, nonRefMarkers, p2do, locs, compo, poseBatch):
            """This function expands the input data X to include non-reference markers and the composite marker.

            :param X: The input data to expand.
            :type X: numpy array
            :param nonRefMarkers: The non-reference markers to include.
            :type nonRefMarkers: numpy array
            :param compo: The composite marker.
            :type compo: numpy array
            :param poseBatch: The pose data for the batch.
            :type poseBatch: numpy array
            :return: The expanded input data.
            :rtype: numpy array"""
            expand_inputs(X, nonRefMarkers, compo, poseBatch)
            p2dflat = np.concatenate(self.project_composite(index=-1, locs=locs))
            return (p2dflat - p2do).flatten().astype(np.float64)

        if len(self.composites) == 0:
            self.graph_calibration()
        self.composites.append(self.composites[from_index].copy())
        self.poseBatches.append(self.poseBatches[from_index].copy())
        self.history.append("optim")
        compo = self.composites[-1]
        poseBatch = self.poseBatches[-1]
        locs = self.markers_on_images()
        p2do = self.observed_points_2D()
        centralMarker = self.centralMarker
        nonRefMarkers = (compo.ids == centralMarker) == False
        args = (nonRefMarkers, p2do, locs, compo, poseBatch)
        X0 = flatten_inputs(nonRefMarkers, compo, poseBatch)
        self.t0 = time.time()
        t = threading.Thread(target=self.print_elapsed_time)
        self._run_flag = True
        t.start()
        sol = optimize.least_squares(
            cost_function,
            X0,
            method="lm",
            args=args,
            ftol=1e-12,
            xtol=1e-12,
            gtol=1e-10,
        )
        self._run_flag = False
        t.join()
        expand_inputs(sol.x, nonRefMarkers, compo, poseBatch)
        return (compo, poseBatch, sol)

        """1 - Estimate the quality of the tvecs of poses estimations [QTvecs]
            2 - Estimate number of markers seen in the image for each composite [QMk]
            3 - Estimate the global reprojection error of composite markers [QReproj]
            Gatherer all the data in the corresponding poseBatch

        :param Ns: Number of random tries use to define the tvecs deviations. Defaults to 50.
        :type Ns: int
        :param frame: The dataframe containing the 2D points.
        :type frame: pandas.DataFrame
        :param key: The key for the batch of poses.
        :type key: str
        :param batch: The batch of poses.
        :type batch: pcp.PoseBatch
        :param kind: random method used, can be "uniform" or "gaussian". Defaults to "uniform".
        :type kind: str
        :return: The standard deviation of the translation vectors.
        :rtype: numpy.ndarray"""

        def _get_deltas(sigma_x, sigma_y, size, kind="gaussian", *args, **kwargs):
            """This function generates random points in 2D space.

            :param kind: The type of distribution to use. Can be 'gaussian' or 'uniform'.
            :type kind: str
            :param size: The number of points to generate.
            :type size: int
            :param sigma_x: The standard deviation of the x coordinate.
            :type sigma_x: float
            :param sigma_y: The standard deviation of the y coordinate.
            :type sigma_y: float
            :return: A tuple containing the x and y coordinates of the generated points.
            :rtype: tuple"""
            if kind == "gaussian":
                dx = np.random.normal(0, sigma_x, size=size)
                dy = np.random.normal(0, sigma_y, size=size)
            elif kind == "uniform":
                dx = np.random.uniform(-sigma_x * 3, sigma_x * 3, size=size)
                dy = np.random.uniform(-sigma_y * 3, sigma_y * 3, size=size)
            return (dx, dy)

        def _get_statistics(Ns, frame, key, batch, **kwargs):
            """This function estimates the composite pose sample.

            :param Ns: The number of samples to estimate.
            :type Ns: int
            :param frame: The frame to estimate the composite pose from.
            :type frame: pcp.core.Frame
            :param batch: The batch to estimate the composite pose from.
            :type batch: pcp.core.Batch
            :param key: The key of the composite to estimate the pose of.
            :type key: str
            :return: The estimated composite pose.
            :rtype: np.array"""
            tvecs = np.zeros((Ns, 3))
            (sigma_x, sigma_y) = frame.p2d_residu.std().values
            pbar = tqdm(range(Ns), leave=False)
            for i in pbar:
                pbar.set_description(f"Estimating composite pose sample: {i}")
                p2d = frame.p2d.copy()
                p2d.reset_index(inplace=True)
                (dx, dy) = _get_deltas(sigma_x, sigma_y, size=p2d.shape[0])
                p2d.x += dx
                p2d.y += dy
                ids2d = p2d.marker.unique()
                Nm = len(ids2d)
                p2d = p2d.loc[:, ("x", "y")].values.reshape(Nm, 4, 2)
                (success, rvec, tvec, sol) = pcp.core.estimate_pose_composite_planar(
                    ids2D=ids2d,
                    points2D=p2d,
                    ids3D=batch.composites[key].ids,
                    points3D=batch.composites[key].p3d(),
                    camera_matrix=batch.camera_matrix,
                    distortion_coefficients=batch.distortion_coefficients,
                )
                tvecs[i] = tvec
            return tvecs.std(axis=0)

        proj_data = self.projected_points().copy()
        proj_data.index.names = ["pose", "marker", "detect", "corners"]
        proj_data.columns = pd.MultiIndex.from_tuples(
            [(c[0], "p2d", c[-1]) for c in proj_data.columns]

        res_data = self.get_reprojection_errors().copy()

        )
        res_data.index.names = ["pose", "marker", "detect", "corners"]
        data_p2d = res_data.join(proj_data)
        for key in self.composites.keys():
            Np = self.poseBatches[key].ids
            if QTvecs:
                tvecs_sigma = np.zeros([Np.shape[0], 3])
            nb_markers = np.zeros(Np.shape[0])
            reproj_errors = np.zeros([Np.shape[0], 2])
            pbar = tqdm(Np)
            for i in pbar:
                pbar.set_description(f"Composite {key} pose {i}")
                frame = data_p2d[key].dropna(subset=[("p2d", "x")]).loc[i]
                nb_markers[i] = frame.groupby("marker").count().shape[0]
                if QTvecs:
                    tvecs_sigma[i] = _get_statistics(
                        Ns=Ns, frame=frame, key=key, batch=self, kind=kind
                    )
                for (ind, dd) in frame.groupby("marker"):
                    polygon = Polygon(points=dd["p2d"].values)
                    frame.loc[ind, "p2d_edges"] = polygon.get_edges()
                    frame.loc[ind, "p2d_perimeter"] = polygon.get_perimeter()
                reproj_errors[i] = (
                    (frame["p2d_residu"] ** 2).sum() / nb_markers[i]

            if QTvecs:
                self.poseBatches[key].add_poses_deviations(tvecs_sigma=tvecs_sigma)
            self.poseBatches[key].add_nb_markers_on_pose(nb_markers=nb_markers)
            self.poseBatches[key].add_reprojection_errors(reproj_errors=reproj_errors)

        """This function optimizes the size of the marker to minimize the error in the distance measurements.

        :param targets: The targets to optimize the marker size for.
        :type targets: list of dicts
        :param compo: The composite containing the marker.
        :type compo: pcp.composite.Composite
        :param parameters: The parameters to use for the optimization.
        :type parameters: pcp.parameters.Parameters
        :param camera_matrix: The camera matrix to use for the optimization.
        :type camera_matrix: numpy.array
        :param distorsion_coefficients: The distorsion coefficients to use for the optimization.
        :type distorsion_coefficients: numpy.array
        :return: The optimized composite.
        :rtype: pcp.composite.Composite"""
        targets = self.targets
        compo = self.composite
        parameters = self.parameters
        camera_matrix = self.camera_matrix
        distorsion_coefficients = self.distorsion_coefficients

        def cost(k):
            """This function calculates the distance between two points.

            :param num1: The first point.
            :type num1: int
            :param num2: The second point.
            :type num2: int
            :return: The distance between num1 and num2.
            :rtype: int"""
            Nt = len(targets)
            res = np.array([])
            compo_opt = compo.copy()
            compo_opt.dimensions = compo.dimensions * k[0]
            compo_opt.tvecs = compo.tvecs * k[0]
            for i in range(Nt):
                out = targets[i]["target_dist"] - pcp.calibration.distance_batches(
                    composite=compo_opt,
                    path_0=targets[i]["path_start"],
                    path_1=targets[i]["path_end"],
                    parameters=parameters,
                    camera_matrix=camera_matrix,
                    distorsion_coefficients=distorsion_coefficients,
                    calib_type=calib_type,
                )
                res = np.concatenate([res, out]).flatten()
            return abs(res)

        k0 = np.ones(2)
